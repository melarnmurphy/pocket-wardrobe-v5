# iOS Auth Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Apple Sign In + email+password authentication to the Pocket Wardrobe iOS app, protecting the main TabView behind a session gate and persisting sessions across launches via the Supabase Swift SDK.

**Architecture:** An `@Observable @MainActor AuthManager` class owns the `SupabaseClient` singleton and all auth methods. `RootView` observes it and routes to `AuthView` (unauthenticated) or the existing `TabView` (authenticated), with a loading state during Keychain session restore. Apple Sign In is bridged from callback-based `ASAuthorizationController` to async/await via a `AppleSignInCoordinator` helper.

**Tech Stack:** SwiftUI (iOS 17+), Supabase Swift SDK v2 (`supabase-swift`), `AuthenticationServices` (Apple Sign In), `@Observable` macro

## Task 3: AppleSignInCoordinator

**Files:**
- Create: `ios/PocketWardrobev5/PocketWardrobev5/Auth/AppleSignInCoordinator.swift`

`ASAuthorizationController` uses delegate callbacks. This coordinator bridges them to async/await using a `CheckedContinuation`.

- [ ] **Step 1: Create AppleSignInCoordinator.swift**

  In Xcode: right-click the `Auth` group (create it if it doesn't exist) → **New File → Swift File** → `AppleSignInCoordinator.swift`.

  ```swift
  // Auth/AppleSignInCoordinator.swift
  import AuthenticationServices
  import UIKit

  enum AppleSignInError: LocalizedError {
      case invalidCredential
      var errorDescription: String? {
          "Apple Sign In returned an invalid credential. Please try again."
      }
  }

  /// Bridges ASAuthorizationController's delegate callbacks to async/await.
  /// Must be kept alive (stored as a property) for the duration of the request
  /// because ASAuthorizationController holds a weak reference to its delegate.
  final class AppleSignInCoordinator: NSObject {
      private var continuation: CheckedContinuation<String, Error>?

      /// Presents the Apple Sign In sheet and returns the raw identity token string.
      func requestCredential() async throws -> String {
          try await withCheckedThrowingContinuation { continuation in
              self.continuation = continuation
              let provider = ASAuthorizationAppleIDProvider()
              let request  = provider.createRequest()
              request.requestedScopes = [.email, .fullName]
              let controller = ASAuthorizationController(authorizationRequests: [request])
              controller.delegate = self
              controller.presentationContextProvider = self
              controller.performRequests()
          }
      }
  }

  extension AppleSignInCoordinator: ASAuthorizationControllerDelegate {
      func authorizationController(
          controller: ASAuthorizationController,
          didCompleteWithAuthorization authorization: ASAuthorization
      ) {
          guard
              let credential = authorization.credential as? ASAuthorizationAppleIDCredential,
              let tokenData  = credential.identityToken,
              let token      = String(data: tokenData, encoding: .utf8)
          else {
              continuation?.resume(throwing: AppleSignInError.invalidCredential)
              continuation = nil
              return
          }
          continuation?.resume(returning: token)
          continuation = nil
      }

      func authorizationController(
          controller: ASAuthorizationController,
          didCompleteWithError error: Error
      ) {
          continuation?.resume(throwing: error)
          continuation = nil
      }
  }

  extension AppleSignInCoordinator: ASAuthorizationControllerPresentationContextProviding {
      func presentationAnchor(for controller: ASAuthorizationController) -> ASPresentationAnchor {
          UIApplication.shared.connectedScenes
              .compactMap { $0 as? UIWindowScene }
              .flatMap(\.windows)
              .first(where: \.isKeyWindow) ?? UIWindow()
      }
  }
  ```

  ```swift
  // Auth/AuthManager.swift
  import Supabase
  import SwiftUI

  @Observable
  @MainActor
  final class AuthManager {
      /// Nil until the Supabase session is established. Nil also means unauthenticated.
      private(set) var session: Session?
      /// True while the SDK is restoring a session from Keychain on launch.
      private(set) var isLoading: Bool = true

      let supabase: SupabaseClient

      /// Held during an in-flight Apple Sign In request to prevent deallocation.
      private var appleCoordinator: AppleSignInCoordinator?

      init() {
          supabase = SupabaseClient(
              supabaseURL: URL(string: Config.supabaseURL)!,
              supabaseKey: Config.supabaseAnonKey
          )
          Task { [weak self] in await self?.listenForAuthChanges() }
      }

      // MARK: - Auth state

      private func listenForAuthChanges() async {
          for await (_, session) in supabase.auth.onAuthStateChange {
              self.session  = session
              self.isLoading = false
          }
      }

      // MARK: - Auth actions

      func signInWithPassword(email: String, password: String) async throws {
          try await supabase.auth.signIn(email: email, password: password)
      }

      func signUp(email: String, password: String) async throws {
          try await supabase.auth.signUp(email: email, password: password)
      }

      func sendPasswordReset(email: String) async throws {
          try await supabase.auth.resetPasswordForEmail(email)
      }

      func signOut() async throws {
          try await supabase.auth.signOut()
      }

      func signInWithApple() async throws {
          let coordinator = AppleSignInCoordinator()
          appleCoordinator = coordinator          // retain for delegate lifetime
          defer { appleCoordinator = nil }
          let idToken = try await coordinator.requestCredential()
          try await supabase.auth.signInWithIdToken(
              credentials: OpenIDConnectCredentials(provider: .apple, idToken: idToken)
          )
      }
  }
  ```

  > **API note:** `supabase.auth.onAuthStateChange` is the async stream in supabase-swift 2.x. If the compiler cannot find it, check the installed SDK version — it may be `supabase.auth.authStateChanges` in some minor versions.

  ```swift
  // Auth/AuthView.swift
  import SwiftUI
  import AuthenticationServices

  struct AuthView: View {
      @Environment(AuthManager.self) private var authManager

      enum Mode { case signIn, signUp }

      @State private var mode: Mode           = .signIn
      @State private var email: String        = ""
      @State private var password: String     = ""
      @State private var confirmPassword: String = ""
      @State private var errorMessage: String?
      @State private var notice: String?
      @State private var isBusy: Bool         = false

      var body: some View {
          ScrollView {
              VStack(alignment: .leading, spacing: 0) {

                  // ── Wordmark ────────────────────────────────────────────
                  Text("Pocket Wardrobe")
                      .font(PWFont.display(size: 28))
                      .foregroundStyle(PWColor.ink)
                      .padding(.top, 72)
                      .padding(.bottom, 8)

                  Text(mode == .signIn ? "Sign in to your wardrobe." : "Create your wardrobe.")
                      .font(PWFont.body(size: 15))
                      .foregroundStyle(PWColor.ink60)
                      .padding(.bottom, 40)

                  // ── Mode toggle ─────────────────────────────────────────
                  HStack(spacing: 0) {
                      modeButton("Sign In",    for: .signIn)
                      modeButton("Sign Up",    for: .signUp)
                  }
                  .padding(.bottom, 32)

                  // ── Error banner ────────────────────────────────────────
                  if let errorMessage {
                      Text(errorMessage)
                          .font(PWFont.body(size: 13))
                          .foregroundStyle(PWColor.oxblood)
                          .padding(12)
                          .frame(maxWidth: .infinity, alignment: .leading)
                          .background(PWColor.oxblood.opacity(0.06))
                          .clipShape(RoundedRectangle(cornerRadius: 4))
                          .padding(.bottom, 20)
                  }

                  // ── Notice banner ───────────────────────────────────────
                  if let notice {
                      Text(notice)
                          .font(PWFont.body(size: 13))
                          .foregroundStyle(PWColor.moss)
                          .padding(12)
                          .frame(maxWidth: .infinity, alignment: .leading)
                          .background(PWColor.moss.opacity(0.06))
                          .clipShape(RoundedRectangle(cornerRadius: 4))
                          .padding(.bottom, 20)
                  }

                  // ── Fields ──────────────────────────────────────────────
                  PWField("Email", text: $email)
                      .keyboardType(.emailAddress)
                      .textContentType(.emailAddress)
                      .autocapitalization(.none)
                      .padding(.bottom, 16)

                  PWField("Password", text: $password, isSecure: true)
                      .textContentType(mode == .signIn ? .password : .newPassword)
                      .padding(.bottom, mode == .signUp ? 16 : 8)

                  if mode == .signUp {
                      PWField("Confirm Password", text: $confirmPassword, isSecure: true)
                          .textContentType(.newPassword)
                          .padding(.bottom, 8)
                  }

                  // ── Forgot password ─────────────────────────────────────
                  if mode == .signIn {
                      Button("Forgot password?") { Task { await forgotPassword() } }
                          .font(PWFont.body(size: 13))
                          .foregroundStyle(PWColor.ink60)
                          .padding(.bottom, 28)
                  } else {
                      Spacer().frame(height: 28)
                  }

                  // ── Primary CTA ─────────────────────────────────────────
                  PWButton(
                      title: mode == .signIn ? "Sign In" : "Create Account",
                      style: .primary
                  ) {
                      Task { await submitEmailPassword() }
                  }
                  .disabled(isBusy)
                  .frame(maxWidth: .infinity)
                  .padding(.bottom, 24)

                  // ── Divider ─────────────────────────────────────────────
                  HStack {
                      HairlineDivider()
                      Text("or")
                          .font(PWFont.body(size: 12))
                          .foregroundStyle(PWColor.ink40)
                          .padding(.horizontal, 12)
                      HairlineDivider()
                  }
                  .padding(.bottom, 24)

                  // ── Apple Sign In ───────────────────────────────────────
                  SignInWithAppleButton(.signIn) { request in
                      request.requestedScopes = [.email, .fullName]
                  } onCompletion: { _ in
                      // Handled inside AuthManager.signInWithApple()
                  }
                  .signInWithAppleButtonStyle(.black)
                  .frame(height: 50)
                  .clipShape(RoundedRectangle(cornerRadius: 4))
                  .onTapGesture { Task { await appleSignIn() } }
                  // Note: the onCompletion closure above is unused intentionally —
                  // we call signInWithApple() directly to keep all auth logic in AuthManager.
              }
              .padding(.horizontal, PWSpacing.pageGutter)
              .padding(.bottom, 40)
          }
          .background(PWColor.ivory.ignoresSafeArea())
          .scrollDismissesKeyboard(.interactively)
      }

      // MARK: - Mode toggle

      private func modeButton(_ title: String, for target: Mode) -> some View {
          Button(title) { withAnimation(.easeInOut(duration: 0.15)) { mode = target } }
              .font(PWFont.body(size: 14, weight: mode == target ? .medium : .regular))
              .foregroundStyle(mode == target ? PWColor.ink : PWColor.ink40)
              .padding(.vertical, 8)
              .padding(.trailing, 20)
      }

      // MARK: - Actions

      private func submitEmailPassword() async {
          errorMessage  = nil
          notice        = nil
          isBusy        = true
          defer { isBusy = false }

          do {
              if mode == .signIn {
                  try await authManager.signInWithPassword(email: email, password: password)
              } else {
                  guard password == confirmPassword else {
                      errorMessage = "Passwords do not match."
                      return
                  }
                  try await authManager.signUp(email: email, password: password)
                  notice = "Check your email to confirm your account."
              }
          } catch {
              errorMessage = error.localizedDescription
          }
      }

      private func forgotPassword() async {
          errorMessage = nil
          notice       = nil
          guard !email.isEmpty else {
              errorMessage = "Enter your email address above, then tap Forgot password."
              return
          }
          isBusy = true
          defer { isBusy = false }
          do {
              try await authManager.sendPasswordReset(email: email)
              notice = "Password reset link sent. Check your email."
          } catch {
              errorMessage = error.localizedDescription
          }
      }

      private func appleSignIn() async {
          errorMessage = nil
          isBusy = true
          defer { isBusy = false }
          do {
              try await authManager.signInWithApple()
          } catch {
              // ASAuthorization user-cancelled error — silently ignore
              let code = (error as? ASAuthorizationError)?.code
              if code != .canceled {
                  errorMessage = error.localizedDescription
              }
          }
      }
  }

  // MARK: - PWField (local to AuthView — not shared component)

  private struct PWField: View {
      let label: String
      @Binding var text: String
      var isSecure: Bool = false

      init(_ label: String, text: Binding<String>, isSecure: Bool = false) {
          self.label    = label
          self._text    = text
          self.isSecure = isSecure
      }

      var body: some View {
          VStack(alignment: .leading, spacing: 6) {
              Text(label.uppercased())
                  .font(PWFont.body(size: 10, weight: .medium))
                  .tracking(10 * 0.14)
                  .foregroundStyle(PWColor.ink40)

              Group {
                  if isSecure {
                      SecureField("", text: $text)
                  } else {
                      TextField("", text: $text)
                  }
              }
              .font(PWFont.body(size: 15))
              .foregroundStyle(PWColor.ink)
              .padding(.vertical, 12)
              .padding(.horizontal, 0)

              HairlineDivider()
          }
      }
  }

  #Preview {
      AuthView()
          .environment(AuthManager())
  }
  ```
